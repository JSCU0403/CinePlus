--member
drop table member cascade constraints;
create table member(
	member_code varchar2(30) primary key,
	member_id varchar2(30),
	member_pw varchar2(30),
	member_name varchar2(30),
	member_birth varchar2(10),
	member_phone varchar2(15),
	member_email varchar2(50),
	pw_question varchar2(100),
	pw_answer varchar2(100),
	member_marketing_agree varchar2(10);
	member_email_agree varchar2(10);
	member_sms_agree varchar2(10);
	member_start varchar2(10)
);

--nonmember
drop table nonmember cascade constraints;
create table nonmember (
nonmember_code varchar2(5) Primary key,
nonmember_name varchar2(20) Not Null,
nonmember_phone varchar2(20) Not Null,
nonmember_birth varchar2(20) Not Null,
nonmember_pw varchar2(20) Not Null,
nonmember_agree varchar2(10) Not Null
);

--theater
drop table theater cascade constraints;
create table theater (
theater_code varchar2(5) Primary key,
location_code varchar2(20) Not Null,
screen_code varchar2(20) Not Null,
address varchar2(50) Not Null,
bus varchar2(100) Not Null,
subway varchar2(100) Not Null
);


--location
drop table location cascade constraints;
create table location (
location_code varchar2(10) Primary key,
area varchar2(30) not null,
branch varchar2(50) not null
);
